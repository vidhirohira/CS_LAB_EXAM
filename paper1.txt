// repeat chars
printf "AAAAAAAAAAAAAAAA" > repeat1.txt
printf "AAAAAAAAAAAAAAAA" >> repeat1.txt
openssl enc -aes-128-ecb -e -in repeat1.txt -out repeat_encrypted.enc -K 00112233445566778899aabbccddeeff -nopad
xxd repeat_encrypted.enc
